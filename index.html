import { serve } from 'https://deno.land/std@0.168.0/http/server.ts' import { createClient } from 'https://esm.sh/@supabase/supabase-js@2' const ANTHROPIC_API_KEY = Deno.env.get('ANTHROPIC_API_KEY') ?? '' const SUPABASE_URL = Deno.env.get('SUPABASE_URL') ?? '' const SUPABASE_SERVICE_KEY = Deno.env.get('SUPABASE_SERVICE_ROLE_KEY') ?? '' const PLAN_LIMITS: Record = { free: 3_000, starter: 50_000, pro: 200_000, } const cors = { 'Access-Control-Allow-Origin': '*', 'Access-Control-Allow-Headers': 'authorization, x-client-info, apikey, content-type', 'Access-Control-Allow-Methods': 'POST, OPTIONS', } function json(data: unknown, status = 200) { return new Response(JSON.stringify(data), { status, headers: { ...cors, 'Content-Type': 'application/json' }, }) } function decodeJwt(token: string): Record | null { try { const parts = token.split('.') if (parts.length !== 3) return null const padded = parts[1] + '='.repeat((4 - parts[1].length % 4) % 4) return JSON.parse(atob(padded.replace(/-/g, '+').replace(/_/g, '/'))) } catch { return null } } // Valid Anthropic model names const VALID_MODELS = [ 'claude-sonnet-4-6', 'claude-opus-4-6', 'claude-haiku-4-5-20251001', ] serve(async (req) => { if (req.method === 'OPTIONS') return new Response('ok', { headers: cors }) try { // 1. Auth const token = (req.headers.get('Authorization') ?? '').replace('Bearer ', '').trim() if (!token) return json({ error: 'Chưa đăng nhập' }, 401) const payload = decodeJwt(token) if (!payload) return json({ error: 'Token không hợp lệ' }, 401) const userId = payload.sub as string if (!userId) return json({ error: 'Token thiếu user id' }, 401) const exp = payload.exp as number if (exp && Date.now() / 1000 > exp) { return json({ error: 'Token hết hạn. Đăng nhập lại.' }, 401) } // 2. Admin client const admin = createClient(SUPABASE_URL, SUPABASE_SERVICE_KEY, { auth: { persistSession: false, autoRefreshToken: false }, }) // 3. Get/init user settings const today = new Date().toISOString().split('T')[0] let { data: s } = await admin.from('user_settings').select('*').eq('user_id', userId).single() if (!s) { const { data: ns } = await admin.from('user_settings').insert({ user_id: userId }).select().single() s = ns } if (!s) return json({ error: 'Lỗi cài đặt tài khoản' }, 500) if (s.last_reset_date !== today) { await admin.from('user_settings') .update({ tokens_used_today: 0, last_reset_date: today }) .eq('user_id', userId) s.tokens_used_today = 0 } const plan = s.plan || 'free' const limit = PLAN_LIMITS[plan] ?? 3_000 const used = s.tokens_used_today || 0 if (used >= limit) { return json({ error: 'quota_exceeded', message: `Hết quota ngày (${used}/${limit} tokens). Reset lúc 0:00.`, used, limit, plan }, 429) } // 4. Parse request const body = await req.json().catch(() => ({})) const { prompt, maxTokens = 4000, temperature = 0.8, model } = body if (!prompt) return json({ error: 'Thiếu prompt' }, 400) const temp = Math.min(1.0, Math.max(0.0, parseFloat(String(temperature)) || 0.8)) const maxTok = Math.min(Number(maxTokens) || 4000, 4000) // cap at 4000 // Validate + sanitize model name let safeModel = String(model || 'claude-sonnet-4-6') if (!VALID_MODELS.includes(safeModel)) safeModel = 'claude-sonnet-4-6' // 5. Call Anthropic with timeout (25 seconds) const controller = new AbortController() const timeoutId = setTimeout(() => controller.abort(), 25000) let aRes: Response try { aRes = await fetch('https://api.anthropic.com/v1/messages', { method: 'POST', signal: controller.signal, headers: { 'Content-Type': 'application/json', 'x-api-key': ANTHROPIC_API_KEY, 'anthropic-version': '2023-06-01', }, body: JSON.stringify({ model: safeModel, max_tokens: maxTok, temperature: temp, messages: [{ role: 'user', content: String(prompt) }], }), }) } catch (fetchErr) { clearTimeout(timeoutId) // Timeout or network error — DO NOT charge quota const isTimeout = (fetchErr as Error).name === 'AbortError' return json({ error: isTimeout ? 'Timeout: AI phản hồi quá lâu. Vui lòng thử lại.' : 'Lỗi mạng khi gọi AI.' }, 503) } clearTimeout(timeoutId) if (!aRes.ok) { const err = await aRes.json().catch(() => ({})) // DO NOT charge quota on API errors return json({ error: err?.error?.message || `Lỗi Anthropic: ${aRes.status}` }, aRes.status) } const aData = await aRes.json() const result = aData.content?.[0]?.text || '' // Only charge quota if we got a real result if (!result) { return json({ error: 'AI trả về kết quả rỗng. Không trừ quota.' }, 502) } const inTok = aData.usage?.input_tokens || Math.ceil(String(prompt).length / 4) const outTok = aData.usage?.output_tokens || Math.ceil(result.length / 4) const total = inTok + outTok // 6. Update quota ONLY after successful result await admin.from('user_settings') .update({ tokens_used_today: used + total }) .eq('user_id', userId) return json({ result, usage: { input_tokens: inTok, output_tokens: outTok, total, used_today: used + total, limit, plan, remaining: Math.max(0, limit - used - total), }, }) } catch (e) { console.error('Edge error:', e) return json({ error: 'Lỗi server: ' + (e as Error).message }, 500) } })
